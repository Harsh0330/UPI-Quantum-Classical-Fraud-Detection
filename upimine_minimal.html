# Lightweight feature engineering with safe fallbacks and simulated quantum models


def _series_or_default(df, col, default=0):
    if col in df.columns:
        return pd.to_numeric(df[col], errors="coerce").fillna(default)
    return pd.Series(default, index=df.index, dtype=float)


def engineer_minimal(df):
    d = df.copy()
    for c in [
        "amount",
        "hour",
        "txn_velocity",
        "amt_deviation",
        "failed_attempts_last_24h",
        "user_avg_txn_value",
    ]:
        if c in d.columns:
            d[c] = pd.to_numeric(d[c], errors="coerce")

    d["new_device"] = _series_or_default(d, "new_device", 0).astype(int)
    if "ip_location_mismatch" in d.columns:
        d["ip_mismatch"] = _series_or_default(d, "ip_location_mismatch", 0).astype(int)
    else:
        d["ip_mismatch"] = _series_or_default(d, "ip_mismatch", 0).astype(int)

    if "amount" in d.columns and "user_avg_txn_value" in d.columns:
        denom = pd.to_numeric(d["user_avg_txn_value"], errors="coerce").replace(
            0, np.nan
        )
        d["amount_to_avg"] = pd.to_numeric(d["amount"], errors="coerce") / denom
    else:
        d["amount_to_avg"] = 0.0

    d["risk_flag_count"] = (
        d["new_device"].fillna(0).astype(int)
        + d["ip_mismatch"].fillna(0).astype(int)
        + _series_or_default(d, "night_txn", 0).astype(int)
        + (_series_or_default(d, "failed_attempts_last_24h", 0) >= 3).astype(int)
    )
    d["high_velocity"] = (
        _series_or_default(d, "txn_velocity", 0)
        > _series_or_default(d, "txn_velocity", 0).quantile(0.9)
    ).astype(int)

    for col in ["txn_type", "app", "device"]:
        if col in d.columns:
            d[col] = LabelEncoder().fit_transform(d[col].fillna("UNK").astype(str))
    return d

def stratified_sample_indices(y_values, n_total, seed):
    y_array = np.asarray(y_values)
    positive_idx = np.flatnonzero(y_array == 1)
    negative_idx = np.flatnonzero(y_array == 0)
    rng = np.random.default_rng(seed)
    positive_target = min(max(2, n_total // 4), len(positive_idx))
    negative_target = min(n_total - positive_target, len(negative_idx))
    chosen = np.concatenate(
        [
            rng.choice(positive_idx, size=positive_target, replace=False),
            rng.choice(negative_idx, size=negative_target, replace=False),
        ]
    )
    rng.shuffle(chosen)
    return chosen


def quantum_angles(X_block):
    X_array = np.asarray(X_block, dtype=float)
    if X_array.ndim == 1:
        X_array = X_array.reshape(-1, 1)
    if X_array.shape[1] < 2:
        X_array = np.column_stack([X_array[:, 0], np.zeros(len(X_array))])
    else:
        X_array = X_array[:, :2]
    X_array = np.nan_to_num(X_array)
    scale = np.max(np.abs(X_array), axis=0, keepdims=True) + 1e-9
    return np.pi * (np.tanh(X_array / scale) + 1.0) / 2.0


def quantum_statevectors(X_block):
    angles = quantum_angles(X_block)
    a = angles[:, 0]
    b = angles[:, 1]
    state = np.column_stack(
        [
            np.cos(a / 2) * np.cos(b / 2),
            np.cos(a / 2) * np.sin(b / 2) * np.exp(1j * (a + b) / 2),
            np.sin(a / 2) * np.cos(b / 2) * np.exp(1j * (a - b) / 2),
            np.sin(a / 2) * np.sin(b / 2) * np.exp(1j * (a * b) / 4),
        ]
    )
    norm = np.linalg.norm(state, axis=1, keepdims=True) + 1e-12
    return state / norm


def quantum_kernel_matrix(A, B):
    phi_a = quantum_statevectors(A)
    phi_b = quantum_statevectors(B)
    return np.abs(phi_a @ np.conjugate(phi_b).T) ** 2


def quantum_expectation_features(X_block):
    angles = quantum_angles(X_block)
    a = angles[:, 0]
    b = angles[:, 1]
    z1 = np.cos(a)
    z2 = np.cos(b)
    zz = z1 * z2
    return np.column_stack([z1, z2, zz])

threshold_rows = []
for name, (y_eval, p) in probability_sources.items():
    precision_vals, recall_vals, threshold_vals = precision_recall_curve(y_eval, p)
    f1_vals = (2 * precision_vals * recall_vals) / (
        precision_vals + recall_vals + 1e-12
    )
    best_idx = int(np.nanargmax(f1_vals[:-1])) if len(f1_vals) > 1 else 0
    best_threshold = float(threshold_vals[best_idx]) if len(threshold_vals) else 0.5
